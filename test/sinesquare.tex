\begin{align*}
P_{mon}(t) &= u(t)\cdot i(t) = U\cdot \sin(\omega t) \cdot I\cdot\sin(\omega t) \\
			&= U\cdot I\cdot \sin^2(\omega t)
\end{align*}

\begin{subequations}
\begin{align}
P_{gem}\bigg\rvert_{T} &= \dfrac{1}{T}\int_0^t u(t)\cdot i(t)\,\mathrm{d} t \\
            &= \dfrac{1}{T}\int_0^T U\cdot \sin(\omega t) \cdot I\cdot\sin(\omega t)\,\mathrm{d} t \\
			&= \dfrac{1}{T}\int_0^T U\cdot I\cdot \sin^2(\omega t) \,\mathrm{d} t\\
			&= U\cdot I\cdot \dfrac{1}{T}\int_0^T \sin^2(\omega t) \,\mathrm{d} t\\
			&= U\cdot I\cdot \dfrac{1}{T}\int_0^T \left(\dfrac{1}{2} - \dfrac{1}{2}\cdot\cos(2\omega t)\right)\,\mathrm{d}t \\
			&= \dfrac{U\cdot I}{2\cdot T}\left( \int_0^T 1\,\mathrm{d}t + \int_{0}^{T}\cos(2\omega t)\,\mathrm{d}t\right) \\
			&= \dfrac{U\cdot I}{2\cdot T}\left( T - 0 \right) \\
			&= \dfrac{1}{2}\cdot U \cdot I
\end{align}
\end{subequations}
